WITH i,e,p,v,p1,p2,vv,iN,iM,p.Divisor as pD,CASE WHEN p2 IS null THEN 99 ELSE  toInteger(p2.Eval) END as p2Eval,


///***********************    D4S1     ***********************************/
UNWIND range(2,19) AS n
WITH toString(n) AS N, "20" as nMax

MATCH (v:VertexNode)<-[]-(i:IndexedBy)-[]->(e:Evaluate),(t:TwoSeqFactor)<-[]-(i) where i.N=N AND i.MaxN=nMax AND i.Dimension="2" 
WITH i.N as iN, i.MaxN as iM,e,v,t                        
OPTIONAL MATCH (vv:VertexNode)<-[]-(ii:IndexedBy)-[]->(ee:Evaluate {Value:toString(t.twoSeq)}),(tt:TwoSeqFactor)<-[]-(ii) where ii.MaxN=iM  AND ii.Dimension="2"

WITH iN,iM,e,v,vv,t,tt,ee // index carring and data carring nodes

vI.twoSeq AS vTwoSeq,vI.divisor AS vDivisor,vII.twoSeq AS vvTwoSeq,ee.Value AS eeValue,vII.divisor AS vvDivisor,iN AS N,iM AS MaxN,i.Dimension AS Dimension,v.Scalar AS vScalar,v.Degree AS vDegree,vv.Scalar AS vvScalar,vv.Degree AS vvDegree

CREATE (p:ProductArgsNode {Divisor:"1",Dimension:"2"})         
FOREACH(snoreMe IN CASE  WHEN vv.Vertex IS null  THEN [1] ELSE [] END | 
CREATE (v) <- [p1:ProductRelationT {T:t.twoSeq,Divisor:"2"}] - (p) )
FOREACH(snoreMe IN CASE  WHEN vv.Vertex IS not null  THEN [1] ELSE [] END | 
CREATE (v) <- [p1:ProductRelationT {T:t.twoSeq,Divisor:"2"}] - (p)   
CREATE (p) - [p2:ProductRelationA {T:tt.twoSeq, Eval:ee.Value,Divisor:"2"}] -> (vv)  )
MERGE  (i:IndexedBy {N:iN, Dimension:"4",MaxN:iM,RowCounter:t.twoSeq})
MERGE  (i)-[:VertexIndexedBy]->(p)
MERGE (i) -[:IndexedByEvaluate]->(e)

WITH {MaxN:iM,N:iN,TargetT:t.twoSeq,Tcount:count(t.twoSeq),ArgumentT:tt.twoSeq,Acount:count(tt.twoSeq)} AS gTel
RETURN gTel.MaxN AS MaxN,gTel.N AS N,gTel.TargetT as TargetT,gTel.Tcount AS Tcount,gTel.ArgumentT AS ArgumentT,gTel.Acount as Acount


///****************************    D4S2    ****************************/
UNWIND range(2,14) AS n
WITH toString(n) AS N, "15" as nMax

MATCH (i:IndexedBy)-[]->(e:Evaluate),(i:IndexedBy)-[:VertexIndexedBy]->(p:ProductArgsNode) where i.N=N AND i.MaxN=nMax AND i.Dimension="4"  
WITH i,e,p,i.N as iN, i.MaxN as iM
MATCH (v:VertexNode) <- [p1:ProductRelationT] - (p) 
WITH i,e,p,v,p1,iN,iM,toString( CASE WHEN toString(i.RowCounter)="2" THEN apoc.number.exact.mul(toString(v.Scalar),"2") ELSE toString(v.Scalar) END ) AS TTScalarRowCounter
OPTIONAL MATCH  (p) - [p2:ProductRelationA] -> (vv:VertexNode)



                                                                                                  i,e,p,p1,p2,iN,iM,p.Divisor as pD,
WITH v,vv,CASE WHEN vv.Degree IS null THEN 99 ELSE  toInteger(ee.Value) END as p2Eval,                                                                    
     toString( CASE WHEN toString(v.Degree)="-1" THEN apoc.number.exact.mul(toString(v.Scalar),"2") ELSE toString( CASE WHEN toString(i.RowCounter)="2" THEN apoc.number.exact.mul(toString(v.Scalar),"2") ELSE toString(v.Scalar) END) END ) AS TTScalar,
     toString( CASE WHEN toString(vv.Degree)="-1" THEN apoc.number.exact.mul(toString(vv.Scalar),"2") ELSE apoc.number.exact.mul(toString(vv.Scalar),tt.twoSeq) END) AS AAScalar,     
     toString( CASE WHEN toString(v.Degree)="-1" THEN apoc.number.exact.add(v.Degree,"1") ELSE apoc.number.exact.add(v.Degree,"0") END ) AS TTDegree,
     toString( CASE WHEN toString(vv.Degree)="-1" THEN apoc.number.exact.add(vv.Degree,"1") ELSE apoc.number.exact.add(vv.Degree,"0")  END) AS AADegree

Merge (cl:CollectedProductArgNode {Dimension:'4',MaxN:iM,N:iN,Divisor:"1",afe:p2Eval,tft:toInteger(t.twoSeq)})                                       
CREATE (cl)-[:Arguments]->(aN:ArgumentsNode { TDegree:TTDegree, TScalar:apoc.number.exact.add(TTScalar,"0"), TDivisor:"2", ADivisor:"2",
                                              ATwoSeq:apoc.number.exact.add(tt.twoSeq,"0"), ADegree:AADegree, AScalar:apoc.number.exact.add(AAScalar,"0"),
                                              DegreeSum:CASE WHEN vv.Degree IS Null THEN TTDegree ELSE apoc.number.exact.add(TTDegree,AADegree) END,
                                              ScalarProduct:CASE WHEN vv.Scalar IS Null THEN toString(TTScalar) ELSE apoc.number.exact.mul(TTScalar,AAScalar) END,
                                              ResultDivisor:CASE WHEN vv.Scalar IS Null THEN toString("2") ELSE apoc.number.exact.mul( toString("2"),apoc.number.exact.mul(toString("2"),toString("1"))) END })

WITH {maxN:iM,N:iN,Ncount:count(iN),afe:p2Eval,afeCount:count(p2Eval),tft:toInteger(p1.T),tftCount:count(toInteger(p1.T))} as gs2Tel
RETURN gs2Tel.maxN,gs2Tel.N,gs2Tel.Ncount,gs2Tel.afe,gs2Tel.afeCount,gs2Tel.tft,gs2Tel.tftCount
