MATCH (v:VertexNode)<-[]-(i:IndexedBy)-[]->(e:Evaluate),(t:TwoSeqFactor)<-[]-(i) where i.N='9' AND i.MaxN='10' AND i.Dimension="4"
WITH {DS:collect({Degree:toInteger(v.Degree),Scalar:toFloat(v.Scalar)}),T:toString(t.twoSeq)} as targetFactors,i.N as iN, i.MaxN as iM,e
OPTIONAL MATCH (vv:VertexNode)<-[]-(ii:IndexedBy)-[]->(ee:Evaluate {Value:targetFactors.T}),(tt:TwoSeqFactor)<-[]-(ii) where ii.MaxN='10'  AND ii.Dimension="4"
WITH {DS:Collect({TT:tt.twoSeq,Degree:toInteger(vv.Degree),Scalar:toFloat(vv.Scalar) }),Evaluate:ee.Value}  as argumentFactors  ,targetFactors,iN,iM,e
WITH COLLECT(targetFactors) as targetF,targetFactors.T as tFT,argumentFactors,iN,iM,e 
WITH COLLECT(argumentFactors) as argumentF,argumentFactors.Evaluate as aFE,targetF,tFT,iN,iM,e
UNWIND RANGE(0,SIZE(targetF)-1) as ii
WITH targetF[ii].DS as tFDS, argumentF[ii].DS as aFDS,tFT,aFE,iN,iM,e
UNWIND tFDS as tFDSU
UNWIND aFDS as aFDSU
WITH {afe:aFE,tft:tFT,DS: COLLECT( {  TDegree:CASE WHEN tFDSU.Degree=-1 THEN tFDSU.Degree+1 ELSE tFDSU.Degree END,  // the two case statements are irrelevant; there
                                      TScalar:CASE WHEN tFDSU.Degree=-1 THEN tFDSU.Scalar*2 ELSE tFDSU.Scalar END,  // are no Degree=-1 past dim=2
                                      ATwoSeq:aFDSU.TT,
                                      ADegree:aFDSU.Degree,   
                                      AScalar:CASE WHEN aFDSU.Degree=-1 THEN aFDSU.Scalar*2 ELSE aFDSU.Scalar END })    }  AS argumentProduct,iN,iM,e

WITH COLLECT(argumentProduct) as argumentProductt,argumentProduct.tft as aPafe,iN,iM,e
UNWIND RANGE(0,SIZE(argumentProductt)-1) as iii
WITH argumentProductt[iii].DS as aPTds,aPafe,iN,iM,e
UNWIND aPTds as aPTdsU





WITH CASE WHEN aPTdsU.AScalar IS null THEN 
{DS:COLLECT({ResultScalar:(aPTdsU.TScalar),      //  the /2 is back-spaced out. 
      ResultDegree:aPTdsU.TDegree,
      ResultTwoSeq:aPTdsU.tft}),ET:aPafe}
ELSE
{DS:COLLECT({ResultScalar:(aPTdsU.TScalar*aPTdsU.AScalar)/toInteger(aPafe), 
      ResultDegree: CASE WHEN aPTdsU.ADegree=-1 THEN aPTdsU.TDegree ELSE aPTdsU.ADegree+aPTdsU.TDegree END,
      ResultTwoSeq:aPTdsU.ATwoSeq}),ET:aPafe}
END 
AS resultProduct,iN,iM,e
WITH COLLECT(resultProduct) as rPC,resultProduct.ET as eT,iN,iM,e
UNWIND RANGE(0,SIZE(rPC)-1) as iI
WITH rPC[iI].DS as rPCDS,eT,iN,iM,e
UNWIND rPCDS as rPCDSU
WITH {RScalarList:COLLECT({RScalar:rPCDSU.ResultScalar}),RDegree:rPCDSU.ResultDegree,ET:eT} as resultMap,iN,iM,e
WITH COLLECT(resultMap) as resultMapC,resultMap.ET as eTrm,resultMap.RDegree as rDRM,iN,iM,e
UNWIND RANGE(0,SIZE(resultMapC)-1) as iII
WITH resultMapC[iII].RScalarList as rSL,eTrm,rDRM,iN,iM,e
UNWIND RANGE(0,SIZE(rSL)-1) as iIII
UNWIND rSL[iIII] as rSLI
WITH COLLECT(rSLI.RScalar) as cts,eTrm,rDRM,iN,iM,e
With REDUCE(tSum=0,ff IN cts | tSum +ff) As TotalRScalar,eTrm,rDRM,iN,iM,e
RETURN TotalRScalar,toInteger(eTrm),toInteger(rDRM),toInteger(iN),toInteger(iM)
