WITH argumentProductt[iii].DS as aPTds,aPafe,iN,iM,apD,aPclafe
UNWIND aPTds as aPTdsU
WITH CASE WHEN aPclafe = 99 THEN                                                                                 
{DS:COLLECT({ResultScalar:(toFloat(aPTdsU.TScalar)),      
      ResultDegree:aPTdsU.TDegree,                                        
      ResultTwoSeq:aPTdsU.tft,
      ResultDivisor:toFloat(aPTdsU.TDivisor)}),ET:aPafe,Divisor:apD}             
ELSE
{DS:COLLECT({ResultScalar:(toFloat(aPTdsU.TScalar)*toFloat(aPTdsU.AScalar)),                     
      ResultDegree: CASE WHEN aPTdsU.ADegree=-1 THEN aPTdsU.TDegree ELSE aPTdsU.ADegree+aPTdsU.TDegree END,
      ResultTwoSeq:aPTdsU.ATwoSeq,
      ResultDivisor:toFloat(aPTdsU.TDivisor)*toFloat(aPTdsU.ADivisor)*toFloat(apD)}),ET:aPafe,Divisor:apD}
END
AS resultProduct,iN,iM
WITH COLLECT(resultProduct) as rPC,resultProduct.ET as eT,iN,iM,resultProduct.Divisor as rpD1

WITH rPC[iI].DS as rPCDS,eT,iN,iM,rpD1

      ,RDivisor:rPCDSU.ResultDivisor,ET:eT,Divisor:rpD1} as resultMap,iN,iM

      resultMap.RDegree as rDRM,iN,iM,resultMap.Divisor as rpD2
UNWIND RANGE(0,SIZE(resultMapC)-1) as iII

MERGE (i) -[:IndexedByEvaluate]->(e)
