




MATCH (v:VertexNode)<-[vI:VertexIndexedBy]-(i:IndexedBy)-[]->(e:Evaluate)          
                     where i.N="9" AND i.MaxN="10" AND i.Dimension="8"
WITH i.N as iN, i.MaxN as iM,e,v,vI,i                        
OPTIONAL MATCH (vv:VertexNode)<-[vII:VertexIndexedBy]-(ii:IndexedByArgument)-[]->(ee:Evaluate {Value:toString(vI.twoSeq)})      
                    where ii.MaxN="10"  AND ii.Dimension="8"
WITH iN,iM,e,v,vv,ee,vI,vII
//return iN,iM,e,v,vv,ee,vI,vII
CREATE (p:ProductArgsNode {Divisor:vI.twoSeq,Dimension:"8"}) 
FOREACH(snoreMe IN CASE  WHEN vv.Degree IS null  THEN [1] ELSE [] END | 
CREATE (v) <- [:ProductRelationT {T:vI.twoSeq,Divisor:vI.divisor}] - (p) )                                                      
FOREACH(snoreMe IN CASE  WHEN vv.Degree IS not null  THEN [1] ELSE [] END | 
CREATE (v) <- [:ProductRelationT {T:vI.twoSeq,Divisor:vI.divisor}] - (p)    
CREATE (p) - [:ProductRelationA {T:vII.twoSeq, Eval:ee.Value,Divisor:vII.divisor}] -> (vv)  )
MERGE  (i:IndexedBy {N:iN, Dimension:"8",MaxN:iM,RowCounter:vI.twoSeq})
MERGE  (i)-[:VertexIndexedBy]->(p)
MERGE (i) -[:IndexedByEvaluate]->(e)
RETURN i,p,v,vv,ee,vI,vII
