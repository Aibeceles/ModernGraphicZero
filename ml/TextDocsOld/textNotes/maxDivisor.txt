MATCH (cl:CollectedProductArgNode {Dimension:'32',MaxN:nMax,N:N})-[:Arguments]->(aN:ArgumentsNode)

MATCH (cl:CollectedProductArgNode {Dimension:'32',MaxN:nMax,N:N})-[:Arguments]->(aN:ArgumentsNode)
WITH {Divisor:cl.Divisor,afe:cl.afe,tft:cl.tft,DS: COLLECT( {  TDegree:aN.TDegree,
      TScalar:aN.TScalar,TDivisor:aN.TDivisor,ADivisor:aN.ADivisor,ATwoSeq:aN.ATwoSeq,
      ADegree:aN.ADegree,AScalar:aN.AScalar,DegreeSum:aN.DegreeSum,ScalarProduct:aN.ScalarProduct,
      ResultDivisor:aN.ResultDivisor})}  AS argumentProduct,  N as iN, nMax as iM,maxDivisor  // cl.N as iN,cl.MaxN as iM,
WITH COLLECT(argumentProduct) as argumentProductt,argumentProduct.tft as aPafe,iN,iM,argumentProduct.Divisor as apD,argumentProduct.afe AS aPclafe,maxDivisor
UNWIND RANGE(0,SIZE(argumentProductt)-1) as iii
WITH argumentProductt[iii].DS as aPTds,aPafe,iN,iM,apD,aPclafe,maxDivisor
UNWIND aPTds as aPTdsU
//RETURN aPTdsU.ScalarProduct,aPTdsU.TScalar,aPTdsU.AScalar,aPTdsU.TDivisor,aPTdsU.ADivisor,aPTdsU.ADegree,aPTdsU.TDegree,aPTdsU.ResultDivisor,aPafe,iN,iM,apD,aPclafe,maxDivisor
WITH CASE WHEN aPclafe = 99 THEN                                                                                 
{DS:COLLECT({ResultScalar:aPTdsU.ScalarProduct,                                                                               
      ResultDegree:aPTdsU.DegreeSum,
      ResultDivisor:aPTdsU.ResultDivisor}),ET:aPafe,Divisor:apD}                                                                                                               
ELSE
{DS:COLLECT({ResultScalar:aPTdsU.ScalarProduct,                       
    ResultDegree:aPTdsU.DegreeSum,    //CASE WHEN aPTdsU.ADegree=-1 THEN aPTdsU.TDegree ELSE apoc.number.exact.add(aPTdsU.ADegree,aPTdsU.TDegree) END,        
    ResultDivisor:aPTdsU.ResultDivisor}),ET:aPafe,Divisor:apD}
END
AS resultProduct,iN,iM,maxDivisor
//RETURN resultProduct
WITH COLLECT(resultProduct) as rPC,resultProduct.ET as eT,iN,iM,resultProduct.Divisor as rpD1,maxDivisor
UNWIND RANGE(0,SIZE(rPC)-1) as iI
WITH rPC[iI].DS as rPCDS,eT,iN,iM,rpD1,maxDivisor
UNWIND rPCDS as rPCDSU
WITH {RScalarList:COLLECT({RScalar:rPCDSU.ResultScalar}),RDegree:rPCDSU.ResultDegree,RDivisor:rPCDSU.ResultDivisor,ET:eT,Divisor:rpD1} as resultMap,iN,iM,maxDivisor
WITH COLLECT(resultMap) as resultMapC,resultMap.RDivisor as rDivisor,resultMap.ET as eTrm,resultMap.RDegree as rDRM,iN,iM,resultMap.Divisor as rpD2,maxDivisor
UNWIND RANGE(0,SIZE(resultMapC)-1) as iII
WITH resultMapC[iII].RScalarList as rSL,rDivisor,rDRM,iN,iM,rpD2,maxDivisor,eTrm
UNWIND RANGE(0,SIZE(rSL)-1) as iIII
UNWIND rSL[iIII] as rSLI
//RETURN rSLI.RScalar,rDivisor,rDRM,iN,iM,maxDivisor,eTrm
WITH COLLECT(rSLI.RScalar) as cts,rDivisor,rDRM,iN,iM,rpD2,"0" as accumulator,maxDivisor
CREATE (a:Accumulator { acc:"0"})
FOREACH (n IN cts| SET a.acc = apoc.number.exact.add(a.acc,n))  // Divisor (aka rowCounter)
WITH {DSList:COLLECT({Degree:rDRM,Scalar:a.acc}),DivideBy:rDivisor} AS pTerms,iM,iN 
//WITH {DSList:COLLECT({Degree:rDRM,Scalar:apoc.number.exact.mul(a.acc,apoc.number.exact.div(maxDivisor,rDivisor))}),DivideBy:maxDivisor} AS pTerms,iM,iN
//RETURN pTerms
WITH COLLECT(pTerms) as pTermss,iM,iN
UNWIND RANGE(0,SIZE(pTermss)-1) as pT
WITH pTermss[pT].DSList as dSList,iN,iM,pTermss[pT].DivideBy AS pTD
UNWIND dSList as dSListU
//RETURN dSListU.Degree,dSListU.Scalar,pTD
CREATE (c:Accumulator {acc:"1"})   
FOREACH (n IN RANGE(1,toInteger(dSListU.Degree))| SET c.acc = apoc.number.exact.mul(c.acc,toString(iM)))   //  dSListU.Degree must be integer
With apoc.number.exact.mul(dSListU.Scalar,c.acc) as scalarDegreeProduct,pTD,iN,iM
WITH {SDP:COLLECT(scalarDegreeProduct),DivideBy:toString(pTD)} as termsMap,iN,iM
//Return termsMap
WITH COLLECT(termsMap) AS termsMapC,iN,iM
UNWIND RANGE(0,SIZE(termsMapC)-1) as pTT
CREATE (b:Accumulator { acc:"0" ,db:toString(termsMapC[pTT].DivideBy) })
FOREACH (n IN termsMapC[pTT].SDP| SET b.acc = apoc.number.exact.add(b.acc,n))
WITH apoc.number.exact.div(b.acc,toString(b.db)) AS pResult,b,iM,iN

WITH COLLECT(pResult) as evaluates,iN,iM
//return evaluates,iN,iM
CREATE (d:Accumulator { acc:"0"})
FOREACH (n IN evaluates| SET d.acc = apoc.number.exact.add(d.acc,n)) 

RETURN d.acc as Value
