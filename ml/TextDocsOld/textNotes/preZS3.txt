MATCH (cl:CollectedProductArgNode {Dimension:'8',MaxN:'10',N:'9'})-[:Arguments]->(aN:ArgumentsNode)
With max(aN.ResultDivisor) as maxDivisor

MATCH (cl:CollectedProductArgNode {Dimension:'8',MaxN:'10',N:'9'})-[:Arguments]->(aN:ArgumentsNode)
WITH {Divisor:cl.Divisor,afe:cl.afe,tft:cl.tft,DS: COLLECT( {  TDegree:aN.TDegree,
                                      TScalar:aN.TScalar,
                                      TDivisor:aN.TDivisor,
                                      ADivisor:aN.ADivisor,
                                      ATwoSeq:aN.ATwoSeq,
                                      ADegree:aN.ADegree,   
                                      AScalar:aN.AScalar,
                                      ScalarProduct:aN.ScalarProduct,
                                      ResultDivisor:aN.ResultDivisor})}  AS argumentProduct,cl.N as iN,cl.MaxN as iM,maxDivisor
WITH COLLECT(argumentProduct) as argumentProductt,argumentProduct.tft as aPafe,iN,iM,argumentProduct.Divisor as apD,argumentProduct.afe AS aPclafe,maxDivisor
UNWIND RANGE(0,SIZE(argumentProductt)-1) as iii
WITH argumentProductt[iii].DS as aPTds,aPafe,iN,iM,apD,aPclafe,maxDivisor
UNWIND aPTds as aPTdsU
WITH CASE WHEN aPclafe = 99 THEN                                                                                 
{DS:COLLECT({ResultScalar:aPTdsU.ScalarProduct,                                                                          // ResultScalar      
      ResultDegree:aPTdsU.TDegree,                                                                                   // ResultDegree
      //ResultTwoSeq:aPTdsU.tft,
      ResultDivisor:toFloat(aPTdsU.ResultDivisor)}),ET:aPafe,Divisor:apD}                                                            
ELSE
{DS:COLLECT({ResultScalar:aPTdsU.ScalarProduct,                       
    ResultDegree: CASE WHEN aPTdsU.ADegree=-1 THEN aPTdsU.TDegree ELSE toFloat(aPTdsU.ADegree)+toFloat(aPTdsU.TDegree) END,          // ResultDegree compute
    //ResultTwoSeq:aPTdsU.ATwoSeq,
    ResultDivisor:toFloat(aPTdsU.ResultDivisor)}),ET:aPafe,Divisor:apD}
END
AS resultProduct,iN,iM,maxDivisor
WITH COLLECT(resultProduct) as rPC,resultProduct.ET as eT,iN,iM,resultProduct.Divisor as rpD1,maxDivisor
UNWIND RANGE(0,SIZE(rPC)-1) as iI
WITH rPC[iI].DS as rPCDS,eT,iN,iM,rpD1,maxDivisor
UNWIND rPCDS as rPCDSU
WITH {RScalarList:COLLECT({RScalar:rPCDSU.ResultScalar}),RDegree:rPCDSU.ResultDegree,RDivisor:rPCDSU.ResultDivisor,ET:eT,Divisor:rpD1} as resultMap,iN,iM,maxDivisor
WITH COLLECT(resultMap) as resultMapC,resultMap.RDivisor as rDivisor,resultMap.RDegree as rDRM,iN,iM,resultMap.Divisor as rpD2,maxDivisor
UNWIND RANGE(0,SIZE(resultMapC)-1) as iII
WITH resultMapC[iII].RScalarList as rSL,rDivisor,rDRM,iN,iM,rpD2,maxDivisor
UNWIND RANGE(0,SIZE(rSL)-1) as iIII
UNWIND rSL[iIII] as rSLI
WITH COLLECT(rSLI.RScalar) as cts,rDivisor,rDRM,iN,iM,rpD2,"0" as accumulator,maxDivisor
CREATE (a:Accumulator { acc:"0"})
FOREACH (n IN cts| SET a.acc = apoc.number.exact.add(a.acc,n))  
WITH {DSList:COLLECT({Degree:rDRM,Scalar:a.acc}),DivideBy:rDivisor} AS pTerms,iM,iN         //*************************//
//WITH {DSList:COLLECT({Degree:rDRM,Scalar:CASE WHEN maxDivisor/rDivisor > 1 THEN apoc.number.exact.mul(a.acc,toString(maxDivisor/rDivisor)) ELSE a.acc END}),DivideBy:maxDivisor} AS pTerms,iM,iN
WITH COLLECT(pTerms) as pTermss,iM,iN

UNWIND RANGE(0,SIZE(pTermss)-1) as pT                                                        //*************************//
WITH pTermss[pT].DSList as dSList,iN,iM    ,pTermss[pT].DivideBy AS pTD
UNWIND dSList as dSListU
WITH dSListU.Degree AS tDegree,dSListU.Scalar AS tScalar,iN,iM,pTD

MERGE (i:IndexedByArgument {N:iN,MaxN:iM,Dimension:"16"} )  
CREATE (v:VertexNode {Scalar:tScalar,Degree:toString(tDegree)} )
MERGE (e:Evaluate {Value:toString(toInteger(2^toInteger(iN)))}) //  Must be integer value.... Must compute evaluate   or just say it is computing 2^n
MERGE (i) -[:IndexedByEvaluate]->(e)
MERGE  (i)-[eee:VertexIndexedBy {twoSeq:"99", divisor:pTD}]->(v)

RETURN i,v,e
