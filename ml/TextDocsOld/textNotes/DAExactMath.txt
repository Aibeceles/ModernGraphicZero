MATCH (cl:CollectedProductArgNode {Dimension:'16',MaxN:'10',N:'9'})-[:Arguments]->(aN:ArgumentsNode)
WITH {Divisor:cl.Divisor,afe:cl.afe,tft:cl.tft,DS: COLLECT( {  TDegree:aN.TDegree,
                                      TScalar:aN.TScalar,
                                      TDivisor:aN.TDivisor,
                                      ADivisor:aN.ADivisor,
                                      ATwoSeq:aN.ATwoSeq,
                                      ADegree:aN.ADegree,   
                                      AScalar:aN.AScalar})}  AS argumentProduct,cl.N as iN,cl.MaxN as iM
WITH COLLECT(argumentProduct) as argumentProductt,argumentProduct.tft as aPafe,iN,iM,argumentProduct.Divisor as apD,argumentProduct.afe AS aPclafe
UNWIND RANGE(0,SIZE(argumentProductt)-1) as iii
WITH argumentProductt[iii].DS as aPTds,aPafe,iN,iM,apD,aPclafe
UNWIND aPTds as aPTdsU
WITH CASE WHEN aPclafe = 99 THEN                                                                        // multiply the scalars, multiply the degrees, add the degrees.                                                                                 
{DS:COLLECT({ResultScalar:(toFloat(aPTdsU.TScalar)),      
      ResultDegree:aPTdsU.TDegree,                                        
      ResultTwoSeq:aPTdsU.tft,
      ResultDivisor:toFloat(aPTdsU.TDivisor)}),ET:aPafe,Divisor:apD}                                   //resultProduct.Divisor seems to have the same value as resultProduct.ET.
ELSE
{DS:COLLECT({ResultScalar:(toFloat(aPTdsU.TScalar)*toFloat(aPTdsU.AScalar)),                     
      ResultDegree: CASE WHEN aPTdsU.ADegree=-1 THEN aPTdsU.TDegree ELSE aPTdsU.ADegree+aPTdsU.TDegree END,
      ResultTwoSeq:aPTdsU.ATwoSeq,
      ResultDivisor:toFloat(aPTdsU.TDivisor)*toFloat(aPTdsU.ADivisor)*toFloat(apD)}),ET:aPafe,Divisor:apD}
END
AS resultProduct,iN,iM
WITH COLLECT(resultProduct) as rPC,resultProduct.ET as eT,iN,iM,resultProduct.Divisor as rpD1
UNWIND RANGE(0,SIZE(rPC)-1) as iI
WITH rPC[iI].DS as rPCDS,eT,iN,iM,rpD1
UNWIND rPCDS as rPCDSU
WITH {RScalarList:COLLECT({RScalar:rPCDSU.ResultScalar}),RDegree:rPCDSU.ResultDegree,RDivisor:rPCDSU.ResultDivisor,ET:eT,Divisor:rpD1} as resultMap,iN,iM
WITH COLLECT(resultMap) as resultMapC,resultMap.RDivisor as rDivisor,resultMap.ET as eTrm,resultMap.RDegree as rDRM,iN,iM,resultMap.Divisor as rpD2
UNWIND RANGE(0,SIZE(resultMapC)-1) as iII
WITH resultMapC[iII].RScalarList as rSL,rDivisor,eTrm,rDRM,iN,iM,rpD2
UNWIND RANGE(0,SIZE(rSL)-1) as iIII
UNWIND rSL[iIII] as rSLI
WITH COLLECT(rSLI.RScalar) as cts,rDivisor,eTrm,rDRM,iN,iM,rpD2
WITH REDUCE(tSum=0,ff IN cts | tSum +ff) As TotalRScalar,rDivisor,eTrm,rDRM,iN,iM,rpD2
With {Degree:rDRM,RDivisor:rDivisor,Divisor:rpD2,Scalar:Collect(TotalRScalar)} AS pTerms,iM
WITH pTerms.Scalar as pTermsS,pTerms.Degree as pTermsD,pTerms.RDivisor as pTRD ,pTerms.Divisor as pTermsDiv,iM
With REDUCE(tSum=0,ff IN pTermsS | tSum +ff) As TRScalar,pTermsD,pTRD,pTermsDiv,iM
RETURN TRScalar,pTermsD,pTRD,pTermsDiv,toInteger(iM)

LEFT
Returns the first character or characters of a text.
Syntax
LEFT("Text"; Number)
Text is the text where the initial partial words are to be determined.
Number (optional) specifies the number of characters for the start text. If this parameter is not defined, one character is returned.
Example
=LEFT("output";3) returns “out”.

LEN
Returns the length of a string including spaces.
Syntax
LEN("Text")
Text is the text whose length is to be determined.

MID
Returns a text string of a text. The parameters specify the starting position and the number of characters.
Syntax
MID("Text"; Start; Number)
Text is the text containing the characters to extract.
Start is the position of the first character in the text to extract.
Number specifies the number of characters in the part of the text.
Example
=MID("office";2;2) returns ff.


RIGHT
Returns the last character or characters of a text.
Syntax
RIGHT("Text"; Number)
Text is the text of which the right part is to be determined.
Number (optional) is the number of characters from the right part of the text.
Example
=RIGHT("Sun";2) returns un.
