    /** Gradient of the input function given in the task*/
  def gradG(x : Array[Double], h : Double, fPL: List[(String,String,String)], fDL: List[(String,String,String)]) : Array[Double] = {
    val n = x.size
    val z : Array[Double] = Array.fill(n){0}
    val y = x
 //   val g0 = g(x,fPL)
    for(i <- 0 until n by 1){
     // y(i) += h
     // z(i) = (g(y) - g0) / h
      z(i) = gDerivative(y,fDL) 
    }
    z
  }
  
  /** derivative of g */
  
  def gDerivative( x : Array[Double], fDL: List[(String,String,String)]   ) : Double = {

     val zero = new BigDecimal("0")
     var result = new BigDecimal("0")
     val argument = new BigDecimal(x(0))
     for (terms <- fDL)
     {
       val deg = (terms._1).toInt match {    //  edge condition from DifferentiateUDF
           case -1 => 0
           case _    => (terms._1).toInt
       }
       val s= new BigDecimal(terms._2)
       val div= new BigDecimal(terms._3)
       result=result.add(argument.pow(deg).multiply(s.divide(div)))
 
     }
 //    println("derivative result: " + result.toString)
     result.doubleValue()

  }

  }
