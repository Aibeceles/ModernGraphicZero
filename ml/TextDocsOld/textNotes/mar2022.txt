def evalHessian(aF:(nextXYD)=>graphArgsD,bF:(nextXYD)=>graphArgsD,cF:(nextXYD)=>graphArgsD,dF:(nextXYD)=>graphArgsD)(xY:nextXYD): hess = {     
    val a=aF(xY).result;val b=bF(xY).result;val c=cF(xY).result;val d=dF(xY).result;
    val det= 1/((a*d)-(b*c))
//    println("evalHessian det:"+ det)
//    println("evalHessian "+a+" "+b+" "+c+" "+d)
    val polyE=hess( d*det,-b*det,-c*det,a*det)      
    polyE
}
