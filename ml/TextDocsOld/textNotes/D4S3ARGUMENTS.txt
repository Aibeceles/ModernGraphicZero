MATCH (cl:CollectedProductArgNode {Dimension:'4',MaxN:'10',N:'9'})
With max(cl.Divisor) as maxDivisor

MATCH (cl:CollectedProductArgNode {Dimension:'4',MaxN:'10',N:'9'})-[:Arguments]->(aN:ArgumentsNode)
WITH {Divisor:cl.Divisor,afe:cl.afe,tft:cl.tft,DS: COLLECT( {  TDegree:aN.TDegree,
                                      TScalar:toString(aN.TScalar),
                                      ATwoSeq:aN.ATwoSeq,
                                      ADegree:aN.ADegree,   
                                      AScalar:toString(aN.AScalar)})}  AS argumentProduct,cl.N as iN,cl.MaxN as iM,maxDivisor
WITH COLLECT(argumentProduct) as argumentProductt,argumentProduct.tft as aPafe,iN,iM,argumentProduct.Divisor as apD,argumentProduct.afe AS aPclafe,maxDivisor
UNWIND RANGE(0,SIZE(argumentProductt)-1) as iii
WITH argumentProductt[iii].DS as aPTds,aPafe,iN,iM,apD,aPclafe,maxDivisor
UNWIND aPTds as aPTdsU
WITH CASE WHEN aPclafe = 99 THEN                                   //  important when collecting different dimensions.                                                          
{DS:COLLECT({ResultScalar:apoc.number.exact.mul(aPTdsU.TScalar,toString(aPafe)), 
      ResultDegree:aPTdsU.TDegree,
      ResultTwoSeq:aPTdsU.tft}),Divisor:apD}
ELSE
{DS:COLLECT({ResultScalar:apoc.number.exact.mul(apoc.number.exact.mul(aPTdsU.TScalar,aPTdsU.AScalar),toString(aPTdsU.ATwoSeq)), 
      ResultDegree: CASE WHEN aPTdsU.ADegree=-1 THEN aPTdsU.TDegree ELSE aPTdsU.ADegree+aPTdsU.TDegree END,
      ResultTwoSeq:aPTdsU.ATwoSeq}),Divisor:apD}
END
AS resultProduct,iN,iM,maxDivisor
WITH COLLECT(resultProduct) as rPC,iN,iM,resultProduct.Divisor as rpD1,maxDivisor
UNWIND RANGE(0,SIZE(rPC)-1) as iI
WITH rPC[iI].DS as rPCDS,iN,iM,rpD1,maxDivisor
UNWIND rPCDS as rPCDSU
WITH {RScalarList:COLLECT({RScalar:rPCDSU.ResultScalar}),RDegree:rPCDSU.ResultDegree,Divisor:rpD1} as resultMap,iN,iM,maxDivisor
WITH COLLECT(resultMap) as resultMapC,resultMap.RDegree as rDRM,iN,iM,resultMap.Divisor as rpD2,maxDivisor
UNWIND RANGE(0,SIZE(resultMapC)-1) as iII
WITH resultMapC[iII].RScalarList as rSL,rDRM,iN,iM,rpD2,maxDivisor
UNWIND RANGE(0,SIZE(rSL)-1) as iIII
UNWIND rSL[iIII] as rSLI
WITH COLLECT(rSLI.RScalar) as cts,rDRM,iN,iM,rpD2,"0" as accumulator,maxDivisor
//RETURN cts,rDRM,iN,iM,rpD2,maxDivisor
CREATE (a:Accumulator { acc:"0"})
FOREACH (n IN cts| SET a.acc = apoc.number.exact.add(a.acc,n))   //  d32 script says rDivisor, locally is rpD2
//WITH {DSList:COLLECT({Degree:rDRM,Scalar:a.acc}),DivideBy:rpD2} AS pTerms,iM,iN    //  common denominator inside the collect map
WITH {DSList:COLLECT({Scalar:CASE WHEN maxDivisor/rpD2 > 1 THEN apoc.number.exact.mul(a.acc,toString(maxDivisor/rpD2)) ELSE a.acc END}),Degree:rDRM,DivideBy:maxDivisor} AS pTerms,iM,iN
WITH COLLECT(pTerms) as pTermss,iM,iN   
//Return pTermss
UNWIND RANGE(0,SIZE(pTermss)-1) as pT                                   //  need to add again scalar list as now common denominators caused a colletion of scalars at each degree.
WITH pTermss[pT].DSList as dSList,iN,iM,pTermss[pT].DivideBy AS pTD,pTermss[pT].Degree AS pTDegree
CREATE (aa:Accumulator { acc:"0"})
WITH dSList,iN,iM,pTD,pTDegree,aa
FOREACH (n IN dSList | SET aa.acc = apoc.number.exact.add(aa.acc,toString(n.Scalar)))
WITH {Scalar:aa.acc,Degree:pTDegree,DivideBy:pTD} as pTermsss,iN,iM
WITH pTermsss.Scalar as tScalar,pTermsss.Degree as tDegree,pTermsss.DivideBy as Divisor,iN,iM
//return Scalar,Degree,DivideBy,iN,iM

MERGE (i:IndexedByArgument {N:iN,MaxN:iM,Dimension:"8"} )  
CREATE (v:VertexNode {Scalar:tScalar,Degree:toString(tDegree)} )
MERGE (e:Evaluate {Value:toString(toInteger(2^toInteger(iN)))}) //  Must be integer value.... Must compute evaluate   or just say it is computing 2^n
MERGE (i) -[:IndexedByEvaluate]->(e)
MERGE  (i)-[eee:VertexIndexedBy {twoSeq:"99", divisor:Divisor}]->(v)

RETURN i,v,e

      ResultDegree:aPTdsU.DegreeSum,
      ResultDivisor:aPTdsU.ResultDivisor}),ET:aPafe,Divisor:apD}

UNWIND RANGE(0,SIZE(rSL)-1) as iIII

//RETURN rSLI.RScalar,rDivisor,rDRM,iN,iM,maxDivisor                         //,eTrm
WITH COLLECT(rSLI.RScalar) as cts,rDivisor,rDRM,iN,iM,rpD2,"0" as accumulator,maxDivisor    //,eTrm

WITH {DSList:COLLECT({Degree:rDRM,Scalar:apoc.number.exact.mul(a.acc,apoc.number.exact.div(maxDivisor,rDivisor))}),DivideBy:maxDivisor} AS pTerms,iM,iN  //,eTrm
//RETURN pTerms
WITH COLLECT(pTerms) as pTermss,iM,iN     //,eTrm
UNWIND RANGE(0,SIZE(pTermss)-1) as pT
WITH pTermss[pT].DSList as dSList,iN,iM,pTermss[pT].DivideBy AS pTD       //,eTrm
